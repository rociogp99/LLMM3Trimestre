<table id="demo"></table>

<script>
function loadDoc() {
  var xhttp = new XMLHttpRequest();
  xhttp.onreadystatechange = function() {
    if (this.readyState == 4 && this.status == 200) {
      myFunction(this);
    }
  };
  xhttp.open("GET", "http://www.aemet.es/xml/municipios/localidad_21041.xml", true);
  xhttp.send();
}
function myFunction(xml) {
  var i;
  var xmlDoc = xml.responseXML;
  var table="<tr><th>Temperaturas</th><th>Estado del cielo</th><th>Direcci√≥n</th><th>Velocidad del viento</th><th>Humedad relativa</th></tr>";
  var x = xmlDoc.getElementsByTagName("CD");
  for (i = 0; i <x.length; i++) { 
    table += "<tr><td>" +
    x[i].getElementsByTagName("temperatura")[0].childNodes[0].nodeValue +
    "</td><td>" +
    x[i].getElementsByTagName("estado_cielo")[0].childNodes[0].nodeValue +
    "</td><td>" +
    x[i].getElementsByTagName("velocidad")[0].childNodes[0].nodeValue +
    "</td><td>" +
    x[i].getElementsByTagName("humedad_relativa")[0].childNodes[0].nodeValue +
    "</td></tr>";
  }
  document.getElementById("demo").innerHTML = table;
}
</script>
